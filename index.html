export async function onRequest(context) { const { request, env } = context; const corsHeaders = { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Headers': 'Content-Type', 'Content-Type': 'application/json', }; if (request.method === 'OPTIONS') { return new Response('', { status: 200, headers: corsHeaders }); } const url = new URL(request.url); const action = url.searchParams.get('action') || ''; const ticker = (url.searchParams.get('ticker') || '').toUpperCase().trim(); // SECURE KEY CHECK const key = env.AV_API_KEY; if (!key) { return new Response(JSON.stringify({ error: 'AV_API_KEY not set in Dashboard' }), { status: 500, headers: corsHeaders }); } try { if (action === 'test') { const resp = await fetch(`https://www.alphavantage.co/query?function=GLOBAL_QUOTE&symbol=AAPL&apikey=${key}`); const data = await resp.json(); return new Response(JSON.stringify({ ok: !!data['Global Quote'] }), { headers: corsHeaders }); } // Default response if no action matches return new Response(JSON.stringify({ message: "API is running" }), { headers: corsHeaders }); } catch (err) { return new Response(JSON.stringify({ error: err.message }), { status: 500, headers: corsHeaders }); } }
